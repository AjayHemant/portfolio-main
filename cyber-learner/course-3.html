Introduction to Python for Security

Python is the go-to language for cybersecurity professionals due to its extensive library ecosystem and ease of use. This course covers essential Python modules used in ethical hacking and security automation.

Why Python for Hacking?

  • Rich Libraries: Extensive security-focused modules
  • Cross-Platform: Works on all major operating systems
  • Community Support: Large security community
  • Integration: Easy to integrate with existing tools

Socket Programming

Sockets are fundamental for network communication and essential for creating network tools.

import socket

# Create a TCP socket
sock = socket.socket(socket.AF_INET, socket.SOCK_STREAM)

# Connect to a server
sock.connect(('example.com', 80))

# Send HTTP request
request = "GET / HTTP/1.1\r\nHost: example.com\r\n\r\n"
sock.send(request.encode())

# Receive response
response = sock.recv(4096)
print(response.decode())

sock.close()

Socket Applications:

  • Port Scanners: Check open ports on targets
  • Banner Grabbing: Identify services and versions
  • Custom Clients: Interact with network services
  • Reverse Shells: Remote access tools

Requests Library

The requests library simplifies HTTP operations for web application testing.

import requests

response = requests.get('https://httpbin.org/get')

Scapy - Packet Manipulation

Scapy is a powerful packet manipulation library for network analysis and testing.

from scapy.all import *

# Create and send a ping packet
packet = IP(dst="8.8.8.8")/ICMP()
response = sr1(packet, timeout=2)

if response:
    print(f"Ping successful: {response.src}")

# Port scan with SYN packets
def syn_scan(target, port):
    packet = IP(dst=target)/TCP(dport=port, flags="S")
    response = sr1(packet, timeout=1, verbose=0)
    
    if response and response[TCP].flags == 18:  # SYN-ACK
        return True
    return False

# Scan common ports
target = "scanme.nmap.org"
common_ports = [22, 23, 53, 80, 110, 443]

for port in common_ports:
    if syn_scan(target, port):
        print(f"Port {port} is open")

Paramiko - SSH Operations

Paramiko enables SSH connections and operations for remote system access.

import paramiko

# SSH connection
ssh = paramiko.SSHClient()
ssh.set_missing_host_key_policy(paramiko.AutoAddPolicy())

try:
    ssh.connect('target.com', username='user', password='pass')
    
    # Execute command
    stdin, stdout, stderr = ssh.exec_command('ls -la')
    print(stdout.read().decode())
    
    # File transfer
    sftp = ssh.open_sftp()
    sftp.get('/remote/file.txt', '/local/file.txt')
    sftp.put('/local/upload.txt', '/remote/upload.txt')
    
except Exception as e:
    print(f"Connection failed: {e}")
finally:
    ssh.close()

BeautifulSoup - Web Scraping

BeautifulSoup is essential for parsing HTML and extracting information from web pages.

import requests
from bs4 import BeautifulSoup

# Fetch and parse webpage
url = "https://example.com"
response = requests.get(url)
soup = BeautifulSoup(response.content, 'html.parser')

# Extract information
title = soup.find('title').text
print(f"Page title: {title}")

# Find all links
links = soup.find_all('a', href=True)
for link in links:
    print(f"Link: {link['href']}")

# Extract forms (useful for finding login pages)
forms = soup.find_all('form')
for form in forms:
    action = form.get('action', '')
    method = form.get('method', 'GET')
    print(f"Form: {method} {action}")
    
    # Find input fields
    inputs = form.find_all('input')
    for inp in inputs:
        name = inp.get('name', '')
        input_type = inp.get('type', 'text')
        print(f"  Input: {name} ({input_type})")

Cryptography Module

The cryptography library provides tools for encryption, decryption, and hashing.

from cryptography.fernet import Fernet
from cryptography.hazmat.primitives import hashes

# Decrypt data
decrypted_data = cipher_suite.decrypt(encrypted_data)
print(f"Decrypted: {decrypted_data}")

password = b"mypassword"

    length=32,
    salt=salt,
    iterations=100000,
)

Threading for Concurrent Operations

Threading allows multiple operations to run simultaneously, essential for efficient scanning tools.

import threading
import socket
from concurrent.futures import ThreadPoolExecutor

def port_scan(target, port):
    try:
        sock = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
        sock.settimeout(1)
        result = sock.connect_ex((target, port))
        sock.close()
        
        if result == 0:
            print(f"Port {port} is open")
            return port
    except:
        pass
    return None

# Multi-threaded port scanner
def threaded_scan(target, ports):
    open_ports = []
    
    with ThreadPoolExecutor(max_workers=50) as executor:
        futures = [executor.submit(port_scan, target, port) for port in ports]
        
        for future in futures:
            result = future.result()
            if result:
                open_ports.append(result)
    
    return open_ports

# Scan common ports
target = "scanme.nmap.org"
ports = range(1, 1001)
open_ports = threaded_scan(target, ports)
print(f"Open ports: {open_ports}")

Building a Simple Port Scanner

Let's combine the concepts to build a practical security tool.

#!/usr/bin/env python3
import socket
import threading
import argparse
from datetime import datetime

class PortScanner:
    def __init__(self, target, threads=50):
        self.target = target
        self.threads = threads
        self.open_ports = []
        self.lock = threading.Lock()
    
    def scan_port(self, port):
        try:
            sock = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
            sock.settimeout(1)
            result = sock.connect_ex((self.target, port))
            
            if result == 0:
                with self.lock:
                    self.open_ports.append(port)
                    print(f"Port {port}: Open")
            
            sock.close()
        except:
            pass
    
    def scan_range(self, start_port, end_port):
        print(f"Scanning {self.target} from port {start_port} to {end_port}")
        print(f"Starting scan at {datetime.now()}")
        
        threads = []
        for port in range(start_port, end_port + 1):
            thread = threading.Thread(target=self.scan_port, args=(port,))
            threads.append(thread)
            thread.start()
            
            # Limit concurrent threads
            if len(threads) >= self.threads:
                for t in threads:
                    t.join()
                threads = []
        
        # Wait for remaining threads
        for thread in threads:
            thread.join()
        
        print(f"Scan completed at {datetime.now()}")
        print(f"Open ports: {sorted(self.open_ports)}")

# Usage example
if __name__ == "__main__":
    scanner = PortScanner("scanme.nmap.org")
    scanner.scan_range(1, 100)
